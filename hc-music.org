check:
       yt-dlp -J --no-playlist URL | jq '.chapters'
if null then use yt-search-m3u
if list then use yt-album-m3u

yt-dlp -J --no-playlist 'https://www.youtube.com/watch?v=fNk_zzaMoSs' | jq '.chapters'
